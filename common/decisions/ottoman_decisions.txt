la_ultima_cruzada = {
	is_shown = {
		C:TUR ?= this
		NOT = { has_variable = la_ultima_cruzada }
	}

	possible = {
		has_law = law_type:law_state_religion
		has_law = law_type:law_monarchy
		is_at_war = yes
	}

	when_taken = {

		add_loyalists = {
			value = 0.05
			religion = rel:sunni
		}

		add_loyalists = {
			value = 0.1
			religion = rel:shiite
		}

		add_radicals = {
			value = 0.15
			religion =rel:catholic
		}

		add_radicals = {
			value = 0.20
			religion =rel:orthodox
		}

		add_modifier = {
			name = la_ultima_cruzada
			months = 12
			is_decaying = yes
		}

		set_variable = {
			name = la_ultima_cruzada
			months = 120
		}

		add_modifier = {
			name = sponsored_state_jihad
			months = 120
		}
	}

	ai_chance = {
		base = 4
	}
}
